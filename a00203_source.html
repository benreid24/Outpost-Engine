71void MidBrain::processMovement(const UpdateContext& ctx) {
72 // ensure we have a path

112 else { enterState(ShootingToKill); }

134void MidBrain::processShooting(const UpdateContext& ctx) {
135 const auto pickFiringPosition = [this, &ctx]() {
136 const world::Node* node = ctx.world.pickFiringPosition(
137 owner.getPosition(), target->getPosition(), Properties.UnitAiFiringMaxDistance.get());
138 if (!node) {
139 BL_LOG_WARN << "Unit could not find suitable firing position";
140 enterState(Failed);
141 return;
142 }
143 destPos = node->getPosition();
144 enterState(MovingToKill);
145 };
146
147 // check if we have a line and are in range
148 if (!waitingForLowBrain) {
149 const float distance = glm::distance(owner.getPosition(), target->getPosition());
150 if (distance >= Properties.UnitAiFiringMaxDistance.get()) {
151 pickFiringPosition();
152 return;
153 }
154
155 com::Combatant* see = ctx.world.lineOfSightIsClear(owner.getPosition(), target);
156 if (see == target || see->getFaction() != owner.getFaction()) {
157 lowBrain.fireAtTarget(target);
158 waitingForLowBrain = true;
159 waitingTime = 0.f;
160 return;
161 }
162 else {
163 pickFiringPosition();
164 return;
165 }
166 }
167 else {
168 switch (lowBrain.getState()) {
169 case LowBrain::Firing:
170 waitingTime += ctx.dt;
171 if (waitingTime >= Properties.UnitAiFiringLineRecheckTime.get()) {
172 waitingForLowBrain = false;
173 }
174 break;
175
176 case LowBrain::Done:
177 enterState(Done);
178 break;
179
180 case LowBrain::Failed:
181 enterState(Failed);
182 break;
183
184 default:
185 BL_LOG_WARN << "Unit low brain in unexpected state: " << lowBrain.getState();
186 waitingForLowBrain = false;
187 break;
