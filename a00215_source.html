170void World::computeNodeDistances(Node& node) {
171 const float MaxCoverDistance = Node::MaxCoverDistance * getWorldToBoxScale();
172 unsigned int i = 0;
173 for (int angle = 0; angle < 360; angle += 30, ++i) {
174 const float r = bl::math::degreesToRadians(static_cast<float>(angle));
175 const float c = std::cos(r);
176 const float s = std::sin(r);
177 // TODO - consider casting a circle?
178 const auto result = b2World_CastRayClosest(
179 getBox2dWorldId(),
180 {node.position.x * getWorldToBoxScale(), node.position.y * getWorldToBoxScale()},
181 {c * MaxCoverDistance, s * MaxCoverDistance},
182 Collisions::getCoverQueryFilter());
183 if (result.hit) {
184 node.distanceToCover[i] = glm::distance(
185 glm::vec2(result.point.x, result.point.y) / getWorldToBoxScale(), node.position);
186 }
187 else { node.distanceToCover[i] = Node::MaxCoverDistance; }
188 }

232 const Node* best = nullptr;

235 coverWeight = 1.f - coverWeight;
236

353bool World::computePath(const glm::vec2& startPos, const glm::vec2& dst, Path& path) {
354 using Waypoint = Path::Waypoint;
355
356 // perform path finding
357 const auto distanceCb = [](const Waypoint& a, const Waypoint& b) -> int {
358 return static_cast<int>(glm::distance(a.position, b.position));
359 };
360 const auto adjacentNodeCb = [this, &dst](const Waypoint& wp,
361 std::vector<std::pair<Waypoint, int>>& neighbors) {
362 if (wp.worldNode) {
363 neighbors.reserve(wp.worldNode->connectsTo.size() + 1);
364 for (std::uint32_t i : wp.worldNode->connectsTo) {
365 const Node* pn = &nodes[i];
366 if (pn->getOccupier() == bl::ecs::InvalidEntity) {
367 neighbors.emplace_back(
368 Waypoint(pn->position, pn),
369 static_cast<int>(glm::distance(wp.position, pn->position)));
370 }
371 }
372 }
373 else {
374 neighbors.reserve(16);
375 for (const Node& node : nodes) {
376 const float d = glm::distance(wp.position, node.position);
377 if (pathToNodeIsClear(wp.position, node)) {
378 neighbors.emplace_back(Waypoint(node.position, &node), static_cast<int>(d));
379 }
380 }
381 }
382
383 if (pathToPositionIsClear(wp.position, dst)) {
384 neighbors.emplace_back(Waypoint(dst),
385 static_cast<int>(glm::distance(wp.position, dst)));
386 }
387 };
388
389 path.currentWaypoint = 0;
390 return PathFinder::findPath({startPos, getNodeAtPosition(startPos)},
391 {dst, getNodeAtPosition(dst)},
392 adjacentNodeCb,
393 distanceCb,
394 path.waypoints);
395}
