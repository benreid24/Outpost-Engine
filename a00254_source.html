173bool DebugMenu::processEvent(const Event& event) {
174 if (!window->visible()) { return false; }
175
176 auto& game = bl::game::Game::getInstance<core::Game>();
177 auto& engine = game.engine();
178 auto& ecs = engine.ecs();
179 auto& player = engine.getPlayer();
180 core::world::World& world = static_cast<core::world::World&>(player.getCurrentWorld());

182 const auto createEntity = [this, &player, &event, &ecs, &game]() {
183 const int fi = entityFactionSelect->getSelectedOption();
184 if (fi < 0 || fi >= factions.size()) { return; }
185
186 constexpr float Radius = 30.f;
187 const auto newEntity = player.getCurrentWorld().createEntity();
188 auto* transform =
189 ecs.emplaceComponent<bl::com::Transform2D>(newEntity, event.worldPosition());
190 game.renderSystem().addTestGraphicsToEntity(newEntity, Radius, sf::Color::Blue);
191
192 auto bodyDef = b2DefaultBodyDef();
193 auto shapeDef = b2DefaultShapeDef();
194 bodyDef.type = b2_dynamicBody;
195 bodyDef.fixedRotation = true;
196 shapeDef.filter = core::world::Collisions::getUnitFilter();
197
198 ecs.emplaceComponent<bl::com::Hitbox2D>(newEntity, transform, Radius);
199 auto physics = game.physicsSystem().addPhysicsToEntity(newEntity, bodyDef, shapeDef);
200 game.damageSystem().makeMortal(newEntity, factions[fi]->getId(), *physics, 100.f);
201
202 controlling.unit =
203 ecs.emplaceComponent<core::com::Unit>(newEntity, factions[fi]->getId(), *physics);
204 controlling.unit->capabilities().add<unit::Capability::Move>();
205 controlling.unit->capabilities().add<unit::Capability::Rotate>();
206 controlling.unit->capabilities().add<unit::Capability::Shoot>();
207
208 ecs.emplaceComponent<core::com::UnitAI>(newEntity, *controlling.unit);
209
210 onEntityControl(newEntity);
211 };

213 const auto createCover = [this, &event, &world]() -> bool {
214 if (event.source().type == sf::Event::MouseButtonPressed) {
215 if (event.source().mouseButton.button == sf::Mouse::Left) {
216 clickStart = event.worldPosition();
217 dragBox.setHidden(false);
218 dragBox.getTransform().setPosition(clickStart);
219 return true;
220 }
221 }
222 else if (!dragBox.component().isHidden()) {
223 if (event.source().type == sf::Event::MouseButtonReleased) {
224 if (event.source().mouseButton.button == sf::Mouse::Left) {
225 const glm::vec2 size = glm::abs(clickStart - event.worldPosition());
226 world.addCover(clickStart, size, getCoverAngle());
227 dragBox.setHidden(true);
228 return true;
229 }
230 }
231 else if (event.source().type == sf::Event::MouseMoved) {
232 const glm::vec2 size = glm::abs(clickStart - event.worldPosition());
233 dragBox.setSize(size);
234 dragBox.getTransform().setOrigin(size * 0.5f);
235 return true;
236 }
237 else if (event.source().type == sf::Event::MouseWheelScrolled) {
238 if (event.source().mouseWheelScroll.delta > 0.f) {
239 coverAngle->incrementValue(1.f);
240 }
241 else { coverAngle->incrementValue(-1.f); }
242 }
243 }
244 return false;
245 };

247 const auto controlEntity = [this, &event, &ecs]() -> bool {
248 if (event.target()) {
249 auto* unit = ecs.getComponent<com::Unit>(event.target()->getId());
250 if (unit) {
251 controlling.unit = unit;
252 onEntityControl(event.target()->getId());
253 return true;
254 }
255 }
256 return false;
257 };
258
259 const auto killEntity = [&engine, &event, &ecs]() -> bool {
260 if (event.target()) {
261 auto* unit = ecs.getComponent<com::Unit>(event.target()->getId());
262 if (unit) {
263 engine.ecs().destroyEntity(event.target()->getId());
264 return true;
265 }
266 }
267 return false;
268 };
269
270 // update hover entity
271 if (event.source().type == sf::Event::MouseMoved) {
272 const auto* phys = game.physicsSystem().findEntityAtPosition(world, event.worldPosition());
273 if (phys) { hoverIdLabel->setText(phys->getOwner().toString()); }
274 else { hoverIdLabel->setText("<none>"); }
275 }
276
277 // process tools
278 switch (currentTopTab) {
279 case TopTab::World:
280 switch (currentWorldTab) {
281 case WorldTab::Nodes:
282 switch (getCurrentNodeTool()) {
283 case NodeTool::CreatePath:
284 if (event.source().type == sf::Event::MouseButtonPressed) {
285 if (event.source().mouseButton.button == sf::Mouse::Left) {
286 world.addNode(core::world::Node::Path, event.worldPosition());
287 return true;
288 }
289 }
290 break;
291
292 case NodeTool::CreateCover:
293 if (event.source().type == sf::Event::MouseButtonPressed) {
294 if (event.source().mouseButton.button == sf::Mouse::Left) {
295 world.addNode(core::world::Node::Cover, event.worldPosition());
296 return true;
297 }
298 }
299 break;
300
301 case NodeTool::Remove:
302 if (event.source().type == sf::Event::MouseButtonPressed) {
303 if (event.source().mouseButton.button == sf::Mouse::Left) {
304 return world.removeNodeAtPosition(event.worldPosition());
305 }
306 }
307 }
308 break; // WorldTab::Nodes
309
310 case WorldTab::Cover:
311 switch (getCurrentCoverTool()) {
312 case CoverTool::Create:
313 return createCover();
314
315 case CoverTool::Remove:
316 if (event.source().type == sf::Event::MouseButtonPressed) {
317 if (event.source().mouseButton.button == sf::Mouse::Left) {
318 return world.removeCoverAtPosition(event.worldPosition());
319 }
320 }
321 }
322 break; // WorldTab::Cover
323 }
324 break; // TopTab::World
325
326 case TopTab::Entity:
327 switch (getCurrentEntityTool()) {
328 case EntityTool::Create:
329 if (event.source().type == sf::Event::MouseButtonPressed) {
330 if (event.source().mouseButton.button == sf::Mouse::Left) {
331 createEntity();
332 return true;
333 }
334 }
335 break;
336 case EntityTool::Control:
337 if (event.source().type == sf::Event::MouseButtonPressed) {
338 if (event.source().mouseButton.button == sf::Mouse::Left) {
339 return controlEntity();
340 }
341 }
342 case EntityTool::Kill:
343 if (event.source().type == sf::Event::MouseButtonPressed) {
344 if (event.source().mouseButton.button == sf::Mouse::Left) { return killEntity(); }
345 }
346 }
347 break; // TopTab::Entity
348
349 case TopTab::Console:
350 default:
351 return false;
352 }
353
354 return false;
355}

357void DebugMenu::update(float) {
358 if (!window->visible()) { return; }
359
360 if (controlling.unit) {
361 auto* mover = controlling.unit->capabilities().get<unit::Capability::Move>();
362 if (mover) {
363 if (sf::Keyboard::isKeyPressed(sf::Keyboard::W)) {
364 mover->move(unit::able::Move::Forward);
365 }
366 else if (sf::Keyboard::isKeyPressed(sf::Keyboard::S)) {
367 mover->move(unit::able::Move::Backward);
368 }
369 else if (sf::Keyboard::isKeyPressed(sf::Keyboard::A)) {
370 mover->move(unit::able::Move::Left);
371 }
372 else if (sf::Keyboard::isKeyPressed(sf::Keyboard::D)) {
373 mover->move(unit::able::Move::Right);
374 }
375 }
376
377 auto* rotater = controlling.unit->capabilities().get<unit::Capability::Rotate>();
378 if (rotater) {
379 if (sf::Keyboard::isKeyPressed(sf::Keyboard::Q)) {
380 rotater->rotate(unit::able::Rotate::CounterClockwise);
381 }
382 else if (sf::Keyboard::isKeyPressed(sf::Keyboard::E)) {
383 rotater->rotate(unit::able::Rotate::Clockwise);
384 }
385 }
386
387 auto* shooter = controlling.unit->capabilities().get<unit::Capability::Shoot>();
388 if (shooter) {
389 if (sf::Keyboard::isKeyPressed(sf::Keyboard::Space)) { shooter->fire(); }
390 }
391 }
