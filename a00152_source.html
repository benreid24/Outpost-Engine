156 void World::computeNodeDistances(Node& node) {
157  const float MaxCoverDistance = Node::MaxCoverDistance * getWorldToBoxScale();
158  unsigned int i = 0;
159  for (int angle = 0; angle < 360; angle += 30, ++i) {
160  const float r = bl::math::degreesToRadians(static_cast<float>(angle));
161  const float c = std::cos(r);
162  const float s = std::sin(r);
163  // TODO - consider casting a circle?
164  const auto result = b2World_CastRayClosest(
165  getBox2dWorldId(),
166  {node.position.x * getWorldToBoxScale(), node.position.y * getWorldToBoxScale()},
167  {c * MaxCoverDistance, s * MaxCoverDistance},
168  Collisions::getCoverQueryFilter());
169  if (result.hit) {
170  node.distanceToCover[i] = glm::distance(
171  glm::vec2(result.point.x, result.point.y) / getWorldToBoxScale(), node.position);

312 bool World::computePath(const glm::vec2& startPos, const glm::vec2& dst, unit::Path& path) {
313  using Waypoint = unit::Path::Waypoint;
314 
315  // perform path finding
316  const auto distanceCb = [](const Waypoint& a, const Waypoint& b) -> int {
317  return static_cast<int>(glm::distance(a.position, b.position));
318  };
319  const auto adjacentNodeCb = [this, &dst](const Waypoint& wp,
320  std::vector<std::pair<Waypoint, int>>& neighbors) {
321  if (wp.worldNode) {
322  neighbors.reserve(wp.worldNode->connectsTo.size() + 1);
323  for (std::uint32_t i : wp.worldNode->connectsTo) {
324  const Node* pn = &nodes[i];
325  if (pn->getOccupier() == bl::ecs::InvalidEntity) {
326  neighbors.emplace_back(
327  Waypoint(pn->position, pn),
328  static_cast<int>(glm::distance(wp.position, pn->position)));
329  }
330  }
331  }
332  else {
333  neighbors.reserve(16);
334  for (const Node& node : nodes) {
335  const float d = glm::distance(wp.position, node.position);
336  if (pathToNodeIsClear(wp.position, node)) {
337  neighbors.emplace_back(Waypoint(node.position, &node), static_cast<int>(d));
338  }
339  }
340  }
341 
342  if (pathToPositionIsClear(wp.position, dst)) {
343  neighbors.emplace_back(Waypoint(dst),
344  static_cast<int>(glm::distance(wp.position, dst)));
345  }
346  };
347 
348  return PathFinder::findPath({startPos, getNodeAtPosition(startPos)},
349  {dst, getNodeAtPosition(dst)},
350  adjacentNodeCb,
351  distanceCb,
352  path.waypoints);
353 }
