150  const auto createEntity = [this, &player, &event, &ecs, &game]() {
151  constexpr float Radius = 30.f;
152  const auto newEntity = player.getCurrentWorld().createEntity();
153  auto* transform =
154  ecs.emplaceComponent<bl::com::Transform2D>(newEntity, event.worldPosition());
155  game.renderSystem().addTestGraphicsToEntity(newEntity, Radius, sf::Color::Blue);

157  auto bodyDef = b2DefaultBodyDef();

168  controlling.unit->makeMoveable(320.f, 1920.f / 6.f, 270.f, 0.9f);
